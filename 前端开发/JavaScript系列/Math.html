      js
      /**
* 生成随机数：
* min：最小值
* max：最大值
* decimalPlaces：保留n位小数，默认值为0，是整数
*/
function getRandomFloat(min, max, decimalPlaces = 0) {
   	// Math.random() 生成一个 0 到 1 之间的随机小数
    // (max - min) 确定生成随机数的范围
	let rand = Math.random() * (max - min) + min;
    // 计算出 10 的 decimalPlaces 次方，用来确定保留的小数位数
	let power = Math.pow(10, decimalPlaces);
    // 将随机数乘以 power，再除以 power，确保返回的随机数具有指定位数的小数部分
	return Math.floor(rand * power) / power;
}
let randomNum = getRandomFloat(0, 100, 2);
console.log(randomNum);

// 生成随机颜色
function getRandomColor() {
	return `#${Math.random().toString(16).substring(2,8).padEnd(6, '0')}`
}
let randomColor = getRandomColor();
console.log(randomColor);

// 生成随机字符串
function getRandomStr(len = 6) {
	if(len <= 11){
		return Math.random()
			.toString(36)
			.substring(2, 2 + len)
			.padEnd(len, '0')
	}else{
		return getRandomStr(11) + getRandomStr(len - 11);
	}
}
let randomStr = getRandomStr(16);
console.log(randomStr);
