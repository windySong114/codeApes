Function函数

  1. 声明式函数 ​

    • 含义：最常见的函数类型，使用function关键字声明函数。函数可以在代码的任何位置进行声明，并且在执行之前会被预先加载到内存中

    • 定义：

      js
      function add(a, b) {
    return a + b;
}

function convertToTree(data,pid = null,level=1){
	return data.filter(item =>item.pid === pid).map(item =>({
		...item,
		level: level,
		children: convertToTree(data,item.id,level + 1)
	}));
}

// 通过递归修改属性值
function updateValue(data, targetId, newValue) {
	data.forEach(node =>{
		if(node.id === targetId){
			node.name = newValue;//修改目标值
		}
		if(node.children){
			updateValue(node.children, targetId, newValue);// 递归遍历子节点
		}
	})
}
// 修改树形结构数据中 id 为 2 的节点的 name 值
updateValue(treeData,2,"new节点名称");
console.log(treeData)
// 把树形结构数据扁平化
function flattenTree(data){
	let result = [];
	function flatten(node, pid = null){
		let newNode = {...node, pid};
		result.push(newNode);
		if(node.children){
			node.children.forEach(child =>flatten(child, node.id));
			delete newNode.children;
		}
	}
	data.forEach(node =>flatten(node));
	return result;
}
let fletData = flattenTree(treeData)
console.log(fletData); // 输出：与allData相同的数据，但id为2的节点的name值为：new节点名称

  6. 闭包函数 ​

    • 含义：闭包是指在一个函数内部定义的函数，并且内部函数可以访问外部函数的变量。闭包是一种函数的结构，描述了函数如何访问其作用域链中的变量。它是一种函数形式，而不仅仅是一种使用方式

    • 定义：

      js
      function outerFunction() {
    var outerVariable = "I am from outer function";
    function innerFunction() {
    	console.log(outerVariable);
    }
    return innerFunction;
}
var closure = outerFunction();
closure(); // 输出 "I am from outer function"

      // 数据封装
function Person(name, age) {
    let privateName = name;
    let privateAge = age;

    return {
        getName: function() {
            return privateName;
        },
        setName: function(name) {
            privateName = name;
        },
        getAge: function() {
            return privateAge;
        },
        setAge: function(age) {
            if (age > 0) {
                privateAge = age;
            }
        }
    };
}

const person = Person('Alice', 30);
console.log(person.getName()); // 输出：Alice
person.setName('Bob');
console.log(person.getName()); // 输出：Bob

      // 迭代器和生成器
function createIterator(array) {
    let index = 0;
    return {
        next: function() {
            return index < array.length ?
                { value: array[index++], done: false } :
                { done: true };
        }
    };
}

const iterator = createIterator([1, 2, 3]);
console.log(iterator.next()); // 输出：{ value: 1, done: false }
console.log(iterator.next()); // 输出：{ value: 2, done: false }
console.log(iterator.next()); // 输出：{ value: 3, done: false }
console.log(iterator.next()); // 输出：{ done: true }

  7. 回调函数 ​

    • 含义：回调函数是一种函数传递的模式，通常作为参数传递给另一个函数，用于在异步操作完成后执行特定的逻辑。回调函数并不是一种独立的函数形式，而是一种使用方式。你可以使用声明式函数、匿名函数或箭头函数等不同类型的函数作为回调函数

    • 定义：

      js
      function fetchData(url, callback) {
    // 模拟异步请求
    setTimeout(function() {
        var data = "Some fetched data";
        callback(data);
    }, 1000);
}
function processData(data) {
	console.log("Processing data:", data);
}
fetchData("https://example.com/api", processData);
