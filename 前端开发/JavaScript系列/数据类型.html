    • NaN：Not a Number，非数值

      js
      // 把字符串强制转换为数字时，会输出NaN
let num = Number("Hello");
console.log(num); // 输出：NaN
// NaN 之间不相等
console.log(NaN === NaN); // 输出 false
// 检测是否为NaN：isNaN()
console.log(isNaN(NaN)); // 输出 true
console.log(isNaN("Hello")); // 输出 true，将字符串无法转换为数字
console.log(isNaN(520)); // 输出 false，520是一个有效的数字
// 避免NaN
let numStr = "123";
let num = Number(numStr); // 将字符串转换为数字
if (!isNaN(num)) {
    console.log(num); // 输出转换后的数字
} else {
    console.log("Invalid number"); // 如果无法转换，则输出错误消息
}

console.log(obj[nameSymbol] == 'bar'); // 输出：false

let arr = [1, 2, 3, 4, 5];
let [first, , third] = arr;
console.log(first,third); // 输出 1 3

// 剩余操作符
let arr = [1, 2, 3, 4, 5];
let [first, second, ...rest] = arr;
console.log(first,second,rest); // 输出 1 2 [3, 4, 5]

let [first, ...rest] = [];
console.log(first,rest); // 输出：undefined []

let [first, ...rest] = ["windy"];
console.log(first,rest); // 输出："windy" []

// 默认值
let arr = [1];
let [a, b = 2] = arr;
console.log(a); // 输出：1
console.log(b); // 输出：2，因为数组中只有一个元素，所以 b 使用了默认值

// 嵌套解构
let arr = [1, [2, 3], 4];
let [a, [b, c], d] = arr;
console.log(a,b,c,d); // 输出：1 2 3 4

let [a, [b, c], d] = arr;
console.log(a,b,c,d); // 输出：1,2,undefined,4
