  1. 基本介绍 ​

    • 含义：译为承诺，是一种用于处理异步操作的对象

    • 作用：解决回调地狱问题

function getUser(userId, callback) {
    setTimeout(function() {
        const user = {
            id: userId,
            username: 'windy',

        callback(user);
    }, 500);
}

function getUserOrders(userId, callback) {
    setTimeout(function() {
        const orders = ['订单1', '订单2', '订单3'];
        callback(orders);

    setTimeout(function() {
        const saved = true; // 假设保存成功
        if (saved) {
            callback('订单保存成功');
        } else {
            callback('订单保存失败');
        }
    }, 1500);
}

  2. 基本用法 ​

    Promise(参数)：Promise对象是一个构造函数，用来生成Promise实例，它接收一个函数作为参数，该函数的两个参数分别是resolve和reject

    • resolve：将Promise对象的状态从"未完成"变为"成功"

    • reject：将Promise对象的状态从"未完成"变为"失败"

      js
      const promise = new Promise(function(resolve, reject) {});

  3. 实例方法 ​

    • then(参数1，参数2)：是实例状态发生改变时的回调函数，它返回一个新的Promise实例（可以进行链式调用的原因），参数1是resolved状态的回调函数，参数2是rejected状态的回调函数

      js
      let promiseFunc = function(isPass){
	return new Promise((resolve, reject) => {
		if(isPass){
			resolve("成功");
		}else{
			reject("失败");
		}
	});
}
// 走成功流程
promiseFunc(true).then(
    // 参数1：res => {}，参数2：err => {}
	res => {
		return res;
	},
	err => {
		// promise执行的是resolve，此处不会打印
		console.log(err);
	}
).then((res)=>{
    // res是上一个then中return出来的值
	console.log(res); // 输出：成功
}).then((res)=>{
    // 上一个then没有return值，此处的res输出的是undefined
	console.log(res); // 输出：undefined
}).finally(() => {
    console.log("完成"); // 输出：完成
});
// 走失败流程
promiseFunc(false).then(
	res => {
		return res;
	},
	err => {
		console.log(err); // 输出：失败
		return err;
	}
).then((res)=>{
    // res是从上一个then中err里return出来的值
	console.log(res); // 输出：失败
}).finally(() => {
    console.log("完成"); // 输出：完成
});

    console.log("完成"); // 输出：完成

  4. 构造方法 ​

    • all(参数)：等待多个异步操作完成，若都成功，则返回结果集合；若有一个失败（即被 reject），则以第一个失败的原因进行拒绝，会忽略后续 Promise 对象的结果；参数是一个数组（迭代对象），数组成员为 Promise 实例

      js
      let promiseFunc = function(isPass){
	return new Promise((resolve, reject) => {
		if(isPass){
			resolve("成功");
		}else{
			reject("失败");
		}
	});
}
try {
    let p1 = await Promise.all([
        promiseFunc(true), 
        promiseFunc(false), 
        promiseFunc(true)
    ]);
    // Promise.all()中promiseFunc(false)会执行reject()，出现异常，此处不执行
	console.log(p1);
} catch (err) {
	// promiseFunc(false)会执行reject()，出现异常
    console.log(err); // 捕捉错误，输出：失败
}
try {
    let p2 = await Promise.all([
        promiseFunc(true), 
        promiseFunc(true), 
        promiseFunc(true)
    ]);
console.log(p2); // 输出：["成功", "成功", "成功"] 
} catch (err) {
	// Promise.all()正常执行结束，没有出现异常，此处不执行
    console.log(err);
}

    • race(参数)：在多个异步操作中，使用竞速的方式处理异步结果，参数是一个数组（迭代对象），数组成员为 Promise 实例，返回的是率先改变状态的 Promise 实例

      js

	return new Promise((resolve, reject) => {
		setTimeout(()=>{
			if(isPass){
				resolve("成功");
			}else{
				reject("失败");
			}
		},delay)
	});
}
try {
    let p1 = await Promise.race([
        promiseFunc(false,1000), 
        promiseFunc(true), 
        promiseFunc(false)
    ]);
    // Promise.all()中，
	// 第1个promiseFunc(false,1000)延迟1000毫秒执行
	// 第2个promiseFunc(true)延迟500毫秒执行
	// 第3个promiseFunc(false)延迟500毫秒执行
	// 综上，按执行顺序和延迟情况，第2个会率先返回结果状态
	console.log(p1); // 输出：成功
} catch (err) {
	// 第2个正常返回，未执行reject，此处不会执行
    console.log(err);
}

  5. 手写一个Promise ​

    js
    // Promise类
class MyPromise {
	constructor(executor) {
		this.status = 'pending'; // 初始状态为 pending
		this.value = undefined; // 成功时的值
		this.reason = undefined; // 失败时的原因

		this.onResolveCallbacks = []; // 存储成功时的回调函数
		this.onRejectCallbacks = []; // 存储失败时的回调函数

		const resolve = (value) => {
			if (this.status === 'pending') {
				this.status = 'fulfilled';
				this.value = value;
				this.onResolveCallbacks.forEach(callback => callback(this.value));
			}
		};

		const reject = (reason) => {
			if (this.status === 'pending') {
				this.status = 'rejected';
				this.reason = reason;
				this.onRejectCallbacks.forEach(callback => callback(this.reason));
			}
		};

		try {
			executor(resolve, reject); // 执行传入的 executor 函数
		} catch (error) {
			reject(error); // 如果执行过程中抛出异常，则认为是失败
		}
	}

	then(onFulfilled, onRejected) {
		// then 方法接收两个可选的回调函数参数
		return new MyPromise((resolve, reject) => {
			if (this.status === 'fulfilled') {
				// 如果当前状态是成功，则立即执行 onFulfilled
				try {
					const result = onFulfilled ? onFulfilled(this.value) : this.value;
					resolve(result); // 将结果传递给下一个 Promise
				} catch (error) {
					reject(error); // 如果执行过程中抛出异常，则认为是失败
				}
			} else if (this.status === 'rejected') {
				// 如果当前状态是失败，则立即执行 onRejected
				try {
					const result = onRejected ? onRejected(this.reason) : this.reason;
					resolve(result); // 将结果传递给下一个 Promise
				} catch (error) {
					reject(error); // 如果执行过程中抛出异常，则认为是失败
				}
			} else if (this.status === 'pending') {
				// 如果当前状态是 pending，则将回调函数存储起来
				this.onResolveCallbacks.push((value) => {
					try {
						const result = onFulfilled ? onFulfilled(value) : value;
						resolve(result); // 将结果传递给下一个 Promise
					} catch (error) {
						reject(error); // 如果执行过程中抛出异常，则认为是失败
					}
				});

				this.onRejectCallbacks.push((reason) => {
					try {
						const result = onRejected ? onRejected(reason) : reason;
						resolve(result); // 将结果传递给下一个 Promise
					} catch (error) {
						reject(error); // 如果执行过程中抛出异常，则认为是失败
					}
				});
			}
		});
	}

	catch (onRejected) {
		return this.then(null, onRejected); // catch 方法其实是 then 方法的简写
	}
}
// 使用
const promise = new MyPromise((resolve, reject) => {
	// 异步操作
	setTimeout(() => {
		resolve('Success!');
		// reject(new Error('Failed!'));
	}, 1000);
});

  7. async/await ​

    • 描述：是一种基于 Promise 的异步编程解决方案，使用 async 关键字声明的函数会返回一个 Promise 对象，该对象的状态会根据异步操作的结果而改变。在异步函数内部使用 await 关键字可以等待 Promise 对象的解决或拒绝，并将结果返回，返回结果后把后续代码推入微任务队列中

    • 优点：避免了回调嵌套的问题；对于promise，async/await处理 then 的调用链，代码更加简洁；

    • 缺点：await会阻塞代码，滥用会导致性能问题

    • 案例：

      js
      async function async1() {
	console.log(1);
	await async2();
	console.log(2);
}
let async2 = async () => {
	await setTimeout(()=>{
		Promise.resolve().then(()=>{
			console.log(3);
		})
		console.log(4);
	},300)
}
let async3 = async () => {
	Promise.resolve().then(()=>{
		console.log(6);
	})
}
async1();
console.log(7);
async3();

// await关键字的作用：等待异步函数的完成
// 若完成，则把后续的代码推入微队列，
